%%% Algorithm %%%
\usepackage{algorithmic}
\usepackage{algorithm}
\newcommand{\algorithmautorefname}{Algorithm}
\renewcommand{\algorithmiccomment}[1]{\texttt{// #1}}
%-----------------------------------------------------------------------------%

\begin{algorithmic}[1]
	\STATE $\text{output} \gets 0$
	\FOR { $\text{time} = 1 \TO 10$ }
		\STATE $\text{output} \gets \text{output} + \text{time}$
	\ENDFOR
	\RETURN $\text{output}$
\end{algorithmic}
